// Cuando creás una nueva rama se copian todos los archivos del ultimo commit del master.
// Si luego de eso se pushean nuevos commits al master, la nueva rama NO los tendrá
// De la misma manera, todos los commits que se hagan en la nueva rama NO aparecerán en la rama master

// crear nueva branch
git branch nuevaRama

//chequear lista de ramas, la que tiene un * es en la que estamos parados.
git branch --list

//para cambiar de rama hacia la nuevaRama.
git checkout nuevaRama

//crea rama y se mueve hacia ella en un paso, para no hacer los dos pasos anteriores

//Merging...


//Para mergear primero hay que cambiar a la rama master con
git checkout master

//Comando para hacer el merge, se abre el editor Vim para escribr el commit del merge , tipeamos :q y enter para salir.
git merge nuevaRama2

git add .
git commit -m "Se mergeó la rama nuevaRama2 a master"

// la consola dice "Your branch is ahead of origin/master" by 2 commits. Use git push to publick your local commits.
// quiere decir que en el entorno local la rama master esta 2 commits adelantada de origin/master, osea el origen // remoto (GitHub), la rama master.
git push

// ahora se puede eliminar la rama nuevaRama2 ya que ya fue mergeada.
// la elimina del entorno LOCAL
git branch -d nuevaRama2

// si nos fijamos en GitHub, sigue figurando la rama nuevaRama2. Para borrarla se puede hacer desde GitHub o por //consola

git push origin :nuevaRama2
